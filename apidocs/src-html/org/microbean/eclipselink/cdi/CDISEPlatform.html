046/**
047 * A {@link JMXServerPlatformBase} that arranges things such that CDI,
048 * not JNDI, will be used to acquire a {@link TransactionManager} and
049 * {@link MBeanServer}.
050 *
051 * <p>Most users will not use this class directly, but will supply its
052 * fully-qualified name as the value of the <a
053 * href="https://www.eclipse.org/eclipselink/documentation/2.7/jpa/extensions/persistenceproperties_ref.htm#target-server">{@code
054 * eclipselink.target-server} Eclipselink JPA extension property</a>
055 * in a <a
056 * href="https://javaee.github.io/tutorial/persistence-intro004.html#persistence-units">{@code
057 * META-INF/persistence.xml} file</a>.</p>
058 *
059 * <p>For example:</p>
060 *
061 * <blockquote><pre>&lt;property name="eclipselink.target-server"
062 *          value="org.microbean.eclipselink.cdi.CDISEPlatform"/&gt;</pre></blockquote>
063 *
064 * @author <a href="https://about.me/lairdnelson"
065 * target="_parent">Laird Nelson</a>
066 *
067 * @see #getExternalTransactionControllerClass()
068 */
069public class CDISEPlatform extends JMXServerPlatformBase {
070
071
072  /*
073   * Instance fields.
074   */
075  
076
077  private final Executor executor;
078
079  private volatile Instance<MBeanServer> mBeanServerInstance;
080  
081
082  /*
083   * Constructors.
084   */
085
086  
087  /**
088   * Creates a {@link CDISEPlatform}.
089   *
090   * @param session the {@link DatabaseSession} this platform will
091   * wrap; must not be {@code null}
092   *
093   * @see JMXServerPlatformBase#JMXServerPlatformBase(DatabaseSession)
094   */
095  public CDISEPlatform(final DatabaseSession session) {
096    super(session);
097    final CDI<Object> cdi = CDI.current();
098    assert cdi != null;
099    if (cdi.select(TransactionManager.class).isUnsatisfied()) {
100      this.disableJTA();
101    }
102    Instance<Executor> executorInstance = cdi.select(Executor.class, Eclipselink.Literal.INSTANCE);
103    assert executorInstance != null;
104    if (executorInstance.isUnsatisfied()) {
105      executorInstance = cdi.select(Executor.class);
106    }
107    assert executorInstance != null;
108    if (executorInstance.isUnsatisfied()) {
109      this.executor = null;
110    } else {
111      this.executor = executorInstance.get();
112    }
113  }
114
115
116  /*
117   * Instance methods.
118   */
119
120  
121  @Override
122  public boolean isRuntimeServicesEnabledDefault() {
123    Instance<MBeanServer> instance = this.mBeanServerInstance;
124    final boolean returnValue;
125    if (instance == null) {
126      final CDI<Object> cdi = CDI.current();
127      instance = cdi.select(MBeanServer.class, Eclipselink.Literal.INSTANCE);
128      assert instance != null;
129      if (instance.isUnsatisfied()) {
130        instance = cdi.select(MBeanServer.class);
131      }
132      assert instance != null; 
133      if (instance.isUnsatisfied()) {
134        returnValue = false;
135      } else {
136        this.mBeanServerInstance = instance;
137        returnValue = true;
138      }
139    } else {
140      returnValue = !instance.isUnsatisfied();
141    }
142    return returnValue;
143  }
144  
145  @Override
146  public MBeanServer getMBeanServer() {
147    if (this.mBeanServer == null) {
148      final Instance<MBeanServer> instance = this.mBeanServerInstance;
149      assert instance != null;
150      if (!instance.isUnsatisfied()) {
151        this.mBeanServer = instance.get();
152      }
153    }
154    return super.getMBeanServer();
155  }
156  
157  @Override
158  public void launchContainerRunnable(final Runnable runnable) {
159    if (runnable != null && this.executor != null) {
160      this.executor.execute(runnable);
161    } else {
162      super.launchContainerRunnable(runnable);
163    }
164  }
165  
166  /**
167   * Returns a non-{@code null} {@link Class} that extends {@link
168   * org.eclipse.persistence.transaction.AbstractTransactionController}.
169   *
170   * @return a non-{@code null} {@link Class} that extends {@link
171   * org.eclipse.persistence.transaction.AbstractTransactionController}
172   *
173   * @see org.eclipse.persistence.transaction.AbstractTransactionController
174   */
175  @Override
176  public final Class<?> getExternalTransactionControllerClass() {
177    if (this.externalTransactionControllerClass == null) {
178      this.externalTransactionControllerClass = TransactionController.class;
179    }
180    return this.externalTransactionControllerClass;
181  }
182
183  /**
184   * Returns {@link JNDIConnector#UNDEFINED_LOOKUP} when invoked.
185   *
186   * @return {@link JNDIConnector#UNDEFINED_LOOKUP}
187   */
188  @Override
189  public final int getJNDIConnectorLookupType() {
190    return JNDIConnector.UNDEFINED_LOOKUP;
191  }

196   */
197

199  /**
200   * A {@link JTATransactionController} whose {@link
201   * #acquireTransactionManager()} uses CDI, not JNDI, to return a
202   * {@link TransactionManager} instance.
203   *
204   * @author <a href="https://about.me/lairdnelson"
205   * target="_parent">Laird Nelson</a>
206   *
207   * @see #acquireTransactionManager()
208   *
209   * @see JTATransactionController
210   */
211  public static final class TransactionController extends JTATransactionController {
212
213
214    /*
215     * Constructors.
216     */
217
218
219    /**
220     * Creates a new {@link TransactionController}.
221     */
222    public TransactionController() {
223      super();
224    }
225
226
227    /*
228     * Instance methods.
229     */
230    
231
232    /**
233     * Returns a non-{@code null} {@link TransactionManager}.
234     *
235     * @return a non-{@code null} {@link TransactionManager}
236     */
237    @Override
238    protected final TransactionManager acquireTransactionManager() {
239      return CDI.current().select(TransactionManager.class).get();
240    }
241
242  }
243
244  /**
245   * A {@link Qualifier} used to designate various things as being
246   * related to <a href="https://www.eclipse.org/eclipselink/"
247   * target="_parent">Eclipselink</a> in some way.
248   *
249   * <p>The typical end user will apply this annotation to an
250   * implementation of {@link Executor} if she wants that particular
251   * {@link Executor} used by the {@link
252   * CDISEPlatform#launchContainerRunnable(Runnable)} method.</p>
253   *
254   * <p>The {@link Eclipselink} qualifier may also be used to annotate
255   * an implementation of {@link MBeanServer} for use by the {@link
256   * CDISEPlatform#getMBeanServer()} method.</p>
257   *
258   * @author <a href="https://about.me/lairdnelson"
259   * target="_parent">Laird Nelson</a>
260   *
261   * @see CDISEPlatform#launchContainerRunnable(Runnable)
262   *
263   * @see CDISEPlatform#getMBeanServer()
264   */
265  @Documented
266  @Qualifier
267  @Retention(RetentionPolicy.RUNTIME)
268  @Target({ ElementType.FIELD, ElementType.METHOD, ElementType.TYPE })
269  public @interface Eclipselink {    
270
271    /**
272     * An {@link AnnotationLiteral} that implements {@link
273     * Eclipselink}.
274     *
275     * @author <a href="https://about.me/lairdnelson"
276     * target="_parent">Laird Nelson</a>
277     */
278    public static final class Literal extends AnnotationLiteral<Eclipselink> implements Eclipselink {
279
280      /**
281       * The single instance of the {@link Literal} class.
282       */
283      public static final Eclipselink INSTANCE = new Literal();
284      
285      private static final long serialVersionUID = 1L;
286      
287    }
288    
289  }
